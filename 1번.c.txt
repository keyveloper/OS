#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>

int main(int argc, char *argv[])
{
    printf("I am parent (pid: %d)\n", (int) getpid());
    int z = 300;
    printf("z value seted!! z: %d \n", z);

    int child = fork(); 

    if (child < 0) { 
        fprintf(stderr, "fork failed\n");
        exit(1); 

    } else if (child == 0) {
        printf("I am child (pid: %d)\n", (int) getpid());
        int old_z = 150;
        printf("child z = %d", z);
        
        z = 150;
        printf("z reseted!! z: $d -> %d", old_z, z);

        
    } else { 
        z = 90;
        printf("in parent z changed z : %d", z);
    }
    return 0;
}
